    classId int NOT NULL PRIMARY KEY AUTO_INCREMENT,
    name text(2000) NOT NULL
);

    subjectId int NOT NULL PRIMARY KEY AUTO_INCREMENT,
    name text(2000) NOT NULL
);

CREATE TABLE chapters(
    chapterId int NOT NULL PRIMARY KEY AUTO_INCREMENT,
    name text(2000) NOT NULL
);

CREATE TABLE classsubjects(
    classsubjectID int NOT NULL PRIMARY KEY AUTO_INCREMENT,
    classId int NOT NULL,
    subjectId int NOT NULL,
    FOREIGN KEY (classId) REFERENCES classes(classId),
    FOREIGN KEY (subjectId) REFERENCES subjects(subjectId)

    subjectchapterID int NOT NULL PRIMARY KEY AUTO_INCREMENT,
    chapterId int NOT NULL,
    subjectId int NOT NULL,
    FOREIGN KEY (chapterId) REFERENCES chapters(chapterId),
